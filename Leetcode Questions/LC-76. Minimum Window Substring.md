def minWindow(self, s: str, t: str) -> str: 
        count = start = 0
        ds = defaultdict(int)
        dt = defaultdict(int)
        len_t = len(t)
        min_substring = ""
        cur_len = math.inf
        for i in range(len(t)):
            dt[t[i]] += 1
        k = len(dt)
        l = r = -1
        for end in range(len(s)):
            ds[s[end]] += 1

            if s[end] in dt and ds[s[end]] == dt[s[end]]:
                count += 1

            while count >= k:
                if end-start+1 < cur_len:
                    cur_len = end-start+1
                    l = start
                    r = end
                ds[s[start]] -= 1
                if ds[s[start]] < dt[s[start]]:
                    count -= 1
                start += 1

        return 	s[l:r+1]
