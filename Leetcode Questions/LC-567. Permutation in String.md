```python
def checkInclusion(self, s1: str, s2: str) -> bool:
        d1 = defaultdict(int)
        d2 = defaultdict(int)
        
        n = len(s1)
        n2 = len(s2)
        
        if n > n2:
            return False
        
        for i in range(n):
            d1[s1[i]] += 1
            d2[s2[i]] += 1

        for i in range(n, n2):
            if self.dictEquals(d1,d2):
                return True
           
            d2[s2[i]] += 1
            d2[s2[i-n]] -= 1 
        return self.dictEquals(d1,d2)
    
    def dictEquals(self, d1: dict, d2: dict) -> bool:
        for key, val in d1.items():
                if d2[key] != val:
                    return False
        return True

```
