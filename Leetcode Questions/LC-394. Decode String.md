# Implementation
```python
def decodeString(self, s: str) -> str:
	nums = set([str(i) for i in range(10)])
	num_stack = []
	str_stack = ['']
	cur_str = ''
	k = 0
	for c in s:
		if c in nums:
			k = k*10 + int(c)
		elif c == '[':
			num_stack.append(k)
			str_stack.append(cur_str)
			k = 0
			cur_str = ''
		elif c == ']':
			n = num_stack.pop()
			prev_word = str_stack.pop()
			
			for i in range(n):
				prev_word = prev_word + cur_str

			cur_str = prev_word
		else:
			cur_str += c

	return cur_str 
```
