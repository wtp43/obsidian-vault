```python
def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
	max_area = 0
	dir = [[0,1], [1,0], [-1,0], [0,-1]]
	n = len(grid)
	m = len(grid[0])
	
	def dfs(i,j):
		nonlocal cur_area
		if i >= n or i < 0 or j >= m or j < 0 or grid[i][j] == 0:
			return 
		grid[i][j] = 0
		cur_area += 1
		for x, y in dir:
			dfs(i + x, j + y)
	
	for i in range(n):
		for j in range(m):
			if grid[i][j] == 0:
				continue
			cur_area = 0
			dfs(i,j)
			max_area = max(max_area, cur_area)
	return max_area
```
