# Implementation
```python
def findAnagrams(self, s: str, p: str) -> List[int]:
	freq_p = Counter()
	freq_s = Counter()
	n = len(p)
	m = len(s)
	res = []

	if m < n:
		return []

	for i in range(n-1):
		freq_p[p[i]] += 1
		freq_s[s[i]] += 1
	freq_p[p[n-1]] += 1

	for i in range(n-1, len(s)):
		freq_s[s[i]] += 1
		if freq_p == freq_s:
			res.append(i-n+1)
		freq_s[s[i-n+1]] -= 1

	return res
```
