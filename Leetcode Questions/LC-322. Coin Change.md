# Implementation
```python
def coinChange(self, coins: List[int], amount: int) -> int:
	d = [0]+[math.inf]*(amount)
	for c in coins:
		# we start at the amount of the coin
		for i in range(c,amount+1):
				d[i] = min(d[i], d[i-c] + 1)
	
	return d[-1] if d[-1] != math.inf else -1   
```
