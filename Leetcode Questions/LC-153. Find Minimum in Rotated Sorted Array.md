# Implementation
```python
def findMin(self, nums: List[int]) -> int:
        i, j = 0, len(nums)-1
        min_elem = math.inf
        while i <= j:
            mid = i+(j-i)//2
            min_elem = min(min_elem, nums[mid])
            if nums[j] > nums[mid]:
                j = mid-1
            else:
                i = mid+1
        return min_elem
```
