```python
def sortedArrayToBST(self, nums: List[int]) -> Optional[TreeNode]:
        def build_tree(l, r):
            if l > r:
                return None
            mid = (l+r)//2
            root = TreeNode(nums[mid])
            root.left = build_tree(l,mid-1)
            root.right = build_tree(mid+1,r) 
            return root
        return build_tree(0, len(nums)-1)
```
