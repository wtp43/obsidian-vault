## Matrix Traversal using [[BFS]]

```python

dirs = [[0,1], [0,-1], [1,0], [-1,0]]

def bfs(grid: List[List[int]], x: int, y: int):
	q = deque((x, y))
	seen = set([(x, y)])
	n_dir = len(dr)
	r = len(grid)
	c = len(grid[0])
	while q:
		cur = q.popleft()
		if cur in seen:
			continue
		for x, y in dirs:
			i = cur[0] + x
			j = cur[1] + y
			if (i,j) in seen or i < 0 or j < 0 or i >= r or j >= c:
				continue
		 	seen.add((i,j))
		 	q.append((i,j))
```
